namespace = military_station_sbs
country_event = {
	id = military_station_sbs.0
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		is_scope_valid = yes
		is_ai = no
		has_technology = tech_defense_platform_hull_1
		any_owned_starbase = {
			fleet = {
				any_controlled_ship = { is_ship_class = shipclass_military_station }
			}
		}
		any_neighbor_country = {
			merg_is_default_empire = yes
			has_communications = prev
			is_at_war_with = prev
			OR = {
				any_war = {
					any_defender = { is_same_empire = root }
					any_attacker = {
						relative_power = {
							who = root
							category = fleet
							value >= superior
						}
					}
				}
			}
		}
	}
	immediate = {
		random_owned_starbase = {
			limit = {
				fleet = {
					any_controlled_ship = { is_ship_class = shipclass_military_station }
				}
			}
			fleet = {
				random_controlled_ship = {
					limit = { is_ship_class = shipclass_military_station }
					set_disable_at_health = 0
				}
			}
			owner = {
				add_modifier = { modifier = station_sbs }
				country_event = { id = military_station_sbs.11 days = 30 }
			}
		}
	}
}
